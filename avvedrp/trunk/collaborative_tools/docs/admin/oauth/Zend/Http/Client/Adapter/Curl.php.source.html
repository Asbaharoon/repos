 124      /**
 125       * Set the configuration array for the adapter
 126       *
 127       * @throws Zend_Http_Client_Adapter_Exception
 128       * @param  Zend_Config | array $config
 129       * @return Zend_Http_Client_Adapter_Curl
 130       */
 131      public function setConfig($config = array())
 132      {
 133          if ($config instanceof Zend_Config) {
 134              $config = $config->toArray();
 135  
 136          } elseif (! is_array($config)) {
 137              require_once 'Zend/Http/Client/Adapter/Exception.php';
 138              throw new Zend_Http_Client_Adapter_Exception(
 139                  'Array or Zend_Config object expected, got ' . gettype($config)
 140              );
 141          }
 142  
 143          if(isset($config['proxy_user']) && isset($config['proxy_pass'])) {
 144              $this->setCurlOption(CURLOPT_PROXYUSERPWD, $config['proxy_user'].":".$config['proxy_pass']);
 145              unset($config['proxy_user'], $config['proxy_pass']);
 146          }
 147  
 148          foreach ($config as $k => $v) {
 149              $option = strtolower($k);
 150              switch($option) {
 151                  case 'proxy_host':
 152                      $this->setCurlOption(CURLOPT_PROXY, $v);
 153                      break;
 154                  case 'proxy_port':
 155                      $this->setCurlOption(CURLOPT_PROXYPORT, $v);
 156                      break;
 157                  default:
 158                      $this->_config[$option] = $v;
 159                      break;
 160              }
 161          }
 162  
 163          return $this;
 164      }

 192      /**
 193       * Initialize curl
 194       *
 195       * @param  string  $host
 196       * @param  int     $port
 197       * @param  boolean $secure
 198       * @return void
 199       * @throws Zend_Http_Client_Adapter_Exception if unable to connect
 200       */
 201      public function connect($host, $port = 80, $secure = false)
 202      {
 203          // If we're already connected, disconnect first
 204          if ($this->_curl) {
 205              $this->close();
 206          }
 207  
 208          // If we are connected to a different server or port, disconnect first
 209          if ($this->_curl
 210              && is_array($this->_connected_to)
 211              && ($this->_connected_to[0] != $host
 212              || $this->_connected_to[1] != $port)
 213          ) {
 214              $this->close();
 215          }
 216  
 217          // Do the actual connection
 218          $this->_curl = curl_init();
 219          if ($port != 80) {
 220              curl_setopt($this->_curl, CURLOPT_PORT, intval($port));
 221          }
 222  
 223          // Set timeout
 224          curl_setopt($this->_curl, CURLOPT_CONNECTTIMEOUT, $this->_config['timeout']);
 225  
 226          // Set Max redirects
 227          curl_setopt($this->_curl, CURLOPT_MAXREDIRS, $this->_config['maxredirects']);
 228  
 229          if (!$this->_curl) {
 230              $this->close();
 231  
 232              require_once 'Zend/Http/Client/Adapter/Exception.php';
 233              throw new Zend_Http_Client_Adapter_Exception('Unable to Connect to ' .  $host . ':' . $port);
 234          }
 235  
 236          if ($secure !== false) {
 237              // Behave the same like Zend_Http_Adapter_Socket on SSL options.
 238              if (isset($this->_config['sslcert'])) {
 239                  curl_setopt($this->_curl, CURLOPT_SSLCERT, $this->_config['sslcert']);
 240              }
 241              if (isset($this->_config['sslpassphrase'])) {
 242                  curl_setopt($this->_curl, CURLOPT_SSLCERTPASSWD, $this->_config['sslpassphrase']);
 243              }
 244          }
 245  
 246          // Update connected_to
 247          $this->_connected_to = array($host, $port);
 248      }

 250      /**
 251       * Send request to the remote server
 252       *
 253       * @param  string        $method
 254       * @param  Zend_Uri_Http $uri
 255       * @param  float         $http_ver
 256       * @param  array         $headers
 257       * @param  string        $body
 258       * @return string        $request
 259       * @throws Zend_Http_Client_Adapter_Exception If connection fails, connected to wrong host, no PUT file defined, unsupported method, or unsupported cURL option
 260       */
 261      public function write($method, $uri, $httpVersion = 1.1, $headers = array(), $body = '')
 262      {
 263          // Make sure we're properly connected
 264          if (!$this->_curl) {
 265              require_once 'Zend/Http/Client/Adapter/Exception.php';
 266              throw new Zend_Http_Client_Adapter_Exception("Trying to write but we are not connected");
 267          }
 268  
 269          if ($this->_connected_to[0] != $uri->getHost() || $this->_connected_to[1] != $uri->getPort()) {
 270              require_once 'Zend/Http/Client/Adapter/Exception.php';
 271              throw new Zend_Http_Client_Adapter_Exception("Trying to write but we are connected to the wrong host");
 272          }
 273  
 274          // set URL
 275          curl_setopt($this->_curl, CURLOPT_URL, $uri->__toString());
 276  
 277          // ensure correct curl call
 278          $curlValue = true;
 279          switch ($method) {
 280              case Zend_Http_Client::GET:
 281                  $curlMethod = CURLOPT_HTTPGET;
 282                  break;
 283  
 284              case Zend_Http_Client::POST:
 285                  $curlMethod = CURLOPT_POST;
 286                  break;
 287  
 288              case Zend_Http_Client::PUT:
 289                  // There are two different types of PUT request, either a Raw Data string has been set
 290                  // or CURLOPT_INFILE and CURLOPT_INFILESIZE are used.
 291                  if(is_resource($body)) {
 292                      $this->_config['curloptions'][CURLOPT_INFILE] = $body;
 293                  }
 294                  if (isset($this->_config['curloptions'][CURLOPT_INFILE])) {
 295                      // Now we will probably already have Content-Length set, so that we have to delete it
 296                      // from $headers at this point:
 297                      foreach ($headers AS $k => $header) {
 298                          if (preg_match('/Content-Length:\s*(\d+)/i', $header, $m)) {
 299                              if(is_resource($body)) {
 300                                  $this->_config['curloptions'][CURLOPT_INFILESIZE] = (int)$m[1];
 301                              }
 302                              unset($headers[$k]);
 303                          }
 304                      }
 305  
 306                      if (!isset($this->_config['curloptions'][CURLOPT_INFILESIZE])) {
 307                          require_once 'Zend/Http/Client/Adapter/Exception.php';
 308                          throw new Zend_Http_Client_Adapter_Exception("Cannot set a file-handle for cURL option CURLOPT_INFILE without also setting its size in CURLOPT_INFILESIZE.");
 309                      }
 310  
 311                      if(is_resource($body)) {
 312                          $body = '';
 313                      }
 314  
 315                      $curlMethod = CURLOPT_PUT;
 316                  } else {
 317                      $curlMethod = CURLOPT_CUSTOMREQUEST;
 318                      $curlValue = "PUT";
 319                  }
 320                  break;
 321  
 322              case Zend_Http_Client::DELETE:
 323                  $curlMethod = CURLOPT_CUSTOMREQUEST;
 324                  $curlValue = "DELETE";
 325                  break;
 326  
 327              case Zend_Http_Client::OPTIONS:
 328                  $curlMethod = CURLOPT_CUSTOMREQUEST;
 329                  $curlValue = "OPTIONS";
 330                  break;
 331  
 332              case Zend_Http_Client::TRACE:
 333                  $curlMethod = CURLOPT_CUSTOMREQUEST;
 334                  $curlValue = "TRACE";
 335                  break;
 336  
 337              default:
 338                  // For now, through an exception for unsupported request methods
 339                  require_once 'Zend/Http/Client/Adapter/Exception.php';
 340                  throw new Zend_Http_Client_Adapter_Exception("Method currently not supported");
 341          }
 342  
 343          if(is_resource($body) && $curlMethod != CURLOPT_PUT) {
 344              require_once 'Zend/Http/Client/Adapter/Exception.php';
 345              throw new Zend_Http_Client_Adapter_Exception("Streaming requests are allowed only with PUT");
 346          }
 347  
 348          // get http version to use
 349          $curlHttp = ($httpVersion == 1.1) ? CURL_HTTP_VERSION_1_1 : CURL_HTTP_VERSION_1_0;
 350  
 351          // mark as HTTP request and set HTTP method
 352          curl_setopt($this->_curl, $curlHttp, true);
 353          curl_setopt($this->_curl, $curlMethod, $curlValue);
 354  
 355          if($this->out_stream) {
 356              // headers will be read into the response
 357              curl_setopt($this->_curl, CURLOPT_HEADER, false);
 358              curl_setopt($this->_curl, CURLOPT_HEADERFUNCTION, array($this, "readHeader"));
 359              // and data will be written into the file
 360              curl_setopt($this->_curl, CURLOPT_FILE, $this->out_stream);
 361          } else {
 362              // ensure headers are also returned
 363              curl_setopt($this->_curl, CURLOPT_HEADER, true);
 364  
 365              // ensure actual response is returned
 366              curl_setopt($this->_curl, CURLOPT_RETURNTRANSFER, true);
 367          }
 368  
 369          // set additional headers
 370          $headers['Accept'] = '';
 371          curl_setopt($this->_curl, CURLOPT_HTTPHEADER, $headers);
 372  
 373          /**
 374           * Make sure POSTFIELDS is set after $curlMethod is set:
 375           * @link http://de2.php.net/manual/en/function.curl-setopt.php#81161
 376           */
 377          if ($method == Zend_Http_Client::POST) {
 378              curl_setopt($this->_curl, CURLOPT_POSTFIELDS, $body);
 379          } elseif ($curlMethod == CURLOPT_PUT) {
 380              // this covers a PUT by file-handle:
 381              // Make the setting of this options explicit (rather than setting it through the loop following a bit lower)
 382              // to group common functionality together.
 383              curl_setopt($this->_curl, CURLOPT_INFILE, $this->_config['curloptions'][CURLOPT_INFILE]);
 384              curl_setopt($this->_curl, CURLOPT_INFILESIZE, $this->_config['curloptions'][CURLOPT_INFILESIZE]);
 385              unset($this->_config['curloptions'][CURLOPT_INFILE]);
 386              unset($this->_config['curloptions'][CURLOPT_INFILESIZE]);
 387          } elseif ($method == Zend_Http_Client::PUT) {
 388              // This is a PUT by a setRawData string, not by file-handle
 389              curl_setopt($this->_curl, CURLOPT_POSTFIELDS, $body);
 390          }
 391  
 392          // set additional curl options
 393          if (isset($this->_config['curloptions'])) {
 394              foreach ((array)$this->_config['curloptions'] as $k => $v) {
 395                  if (!in_array($k, $this->_invalidOverwritableCurlOptions)) {
 396                      if (curl_setopt($this->_curl, $k, $v) == false) {
 397                          require_once 'Zend/Http/Client/Exception.php';
 398                          throw new Zend_Http_Client_Exception(sprintf("Unknown or erroreous cURL option '%s' set", $k));
 399                      }
 400                  }
 401              }
 402          }
 403  
 404          // send the request
 405          $response = curl_exec($this->_curl);
 406  
 407          // if we used streaming, headers are already there
 408          if(!is_resource($this->out_stream)) {
 409              $this->_response = $response;
 410          }
 411  
 412          $request  = curl_getinfo($this->_curl, CURLINFO_HEADER_OUT);
 413          $request .= $body;
 414  
 415          if (empty($this->_response)) {
 416              require_once 'Zend/Http/Client/Exception.php';
 417              throw new Zend_Http_Client_Exception("Error in cURL request: " . curl_error($this->_curl));
 418          }
 419  
 420          // cURL automatically decodes chunked-messages, this means we have to disallow the Zend_Http_Response to do it again
 421          if (stripos($this->_response, "Transfer-Encoding: chunked\r\n")) {
 422              $this->_response = str_ireplace("Transfer-Encoding: chunked\r\n", '', $this->_response);
 423          }
 424  
 425          // Eliminate multiple HTTP responses.
 426          do {
 427              $parts  = preg_split('|(?:\r?\n){2}|m', $this->_response, 2);
 428              $again  = false;
 429  
 430              if (isset($parts[1]) && preg_match("|^HTTP/1\.[01](.*?)\r\n|mi", $parts[1])) {
 431                  $this->_response    = $parts[1];
 432                  $again              = true;
 433              }
 434          } while ($again);
 435  
 436          // cURL automatically handles Proxy rewrites, remove the "HTTP/1.0 200 Connection established" string:
 437          if (stripos($this->_response, "HTTP/1.0 200 Connection established\r\n\r\n") !== false) {
 438              $this->_response = str_ireplace("HTTP/1.0 200 Connection established\r\n\r\n", '', $this->_response);
 439          }
 440  
 441          return $request;
 442      }
