  35  /**
  36   * Makes an HTTP request to the specified URL
  37   *
  38   * @param string $http_method The HTTP method (GET, POST, PUT, DELETE)
  39   * @param string $url Full URL of the resource to access
  40   * @param string $auth_header (optional) Authorization header
  41   * @param string $postData (optional) POST/PUT request body
  42   * @param bool $returnResponseHeaders True if resp. headers should be returned.
  43   * @return string Response body from the server
  44   */
  45  function send_signed_request($http_method, $url, $auth_header=null,
  46                               $postData=null, $returnResponseHeaders=true) {
  47    $curl = curl_init($url);
  48    curl_setopt($curl, CURLOPT_RETURNTRANSFER, true);
  49    curl_setopt($curl, CURLOPT_FAILONERROR, false);
  50    curl_setopt($curl, CURLOPT_SSL_VERIFYPEER, false);
  51  
  52    if ($returnResponseHeaders) {
  53      curl_setopt($curl, CURLOPT_HEADER, true);
  54    }
  55  
  56    switch($http_method) {
  57      case 'GET':
  58        if ($auth_header) {
  59          curl_setopt($curl, CURLOPT_HTTPHEADER, array($auth_header));
  60        }
  61        break;
  62      case 'POST':
  63        $headers = array('Content-Type: application/atom+xml', $auth_header);
  64        curl_setopt($curl, CURLOPT_HTTPHEADER, $headers);
  65        curl_setopt($curl, CURLOPT_POST, 1);
  66        curl_setopt($curl, CURLOPT_POSTFIELDS, $postData);
  67        break;
  68      case 'PUT':
  69        $headers = array('Content-Type: application/atom+xml', $auth_header);
  70        curl_setopt($curl, CURLOPT_HTTPHEADER, $headers);
  71        curl_setopt($curl, CURLOPT_CUSTOMREQUEST, $http_method);
  72        curl_setopt($curl, CURLOPT_POSTFIELDS, $postData);
  73        break;
  74      case 'DELETE':
  75        $headers = array($auth_header);
  76        curl_setopt($curl, CURLOPT_HTTPHEADER, $headers);
  77        curl_setopt($curl, CURLOPT_CUSTOMREQUEST, $http_method);
  78        break;
  79    }
  80    $response = curl_exec($curl);
  81    if (!$response) {
  82      $response = curl_error($curl);
  83    }
  84    curl_close($curl);
  85    return $response;
  86  }

  88  /**
  89   * Prettifies an XML string into a human-readable and indented work of art.
  90   *
  91   * @param string $xml The XML as a string
  92   * @param boolean $html_output True if the output should be escaped (for HTML)
  93   * @return string A beautifully formatted XML string. *
  94   */
  95  function xml_pp($xml, $html_output=false) {
  96    $xml_obj = new SimpleXMLElement($xml);
  97    $level = 4;  // level to indent tagseleme
  98    $indent = 0; // current indentation level
  99    $pretty = array();
 100  
 101    // get an array containing each XML element
 102    $xml = explode("\n", preg_replace('/>\s*</', ">\n<", $xml_obj->asXML()));
 103  
 104    // shift off opening XML tag if present
 105    if (count($xml) && preg_match('/^<\?\s*xml/', $xml[0])) {
 106      $pretty[] = array_shift($xml);
 107    }
 108  
 109    foreach ($xml as $el) {
 110      if (preg_match('/^<([\w])+[^>\/]*>$/U', $el)) {
 111        // opening tag, increase indent
 112        $pretty[] = str_repeat(' ', $indent) . $el;
 113        $indent += $level;
 114      } else {
 115        if (preg_match('/^<\/.+>$/', $el)) {
 116          $indent -= $level;  // closing tag, decrease indent
 117        }
 118        if ($indent < 0) {
 119          $indent += $level;
 120        }
 121        $pretty[] = str_repeat(' ', $indent) . $el;
 122      }
 123    }
 124    $xml = implode("\n", $pretty);
 125    return ($html_output) ? htmlentities($xml) : $xml;
 126  }

 128  /**
 129   * Prettifies a JSON object.
 130   *
 131   * @param string $json A string formmatted as a JSON object
 132   * @return string Formatted json object for pretty print.
 133   */
 134  function json_pp($json) {
 135    $tab = '  ';
 136    $new_json = '';
 137    $indent_level = 0;
 138    $in_string = false;
 139  
 140    $json_obj = json_decode($json);
 141  
 142    if(!$json_obj) {
 143      return $new_json;
 144    }
 145  
 146    $json = json_encode($json_obj);
 147    $len = strlen($json);
 148  
 149    for($c = 0; $c < $len; $c++) {
 150      $char = $json[$c];
 151      switch($char) {
 152        case '{':
 153        case '[':
 154          if(!$in_string) {
 155            $new_json .= $char . "\n" . str_repeat($tab, $indent_level + 1);
 156            $indent_level++;
 157          } else {
 158            $new_json .= $char;
 159          }
 160          break;
 161        case '}':
 162        case ']':
 163          if(!$in_string) {
 164            $indent_level--;
 165            $new_json .= "\n" . str_repeat($tab, $indent_level) . $char;
 166          } else {
 167            $new_json .= $char;
 168          }
 169          break;
 170        case ',':
 171          if(!$in_string) {
 172            $new_json .= ",\n" . str_repeat($tab, $indent_level);
 173          } else {
 174            $new_json .= $char;
 175          }
 176          break;
 177        case ':':
 178          if(!$in_string) {
 179            $new_json .= ': ';
 180          } else {
 181            $new_json .= $char;
 182          }
 183          break;
 184        case '"':
 185          $in_string = !$in_string;
 186        default:
 187          $new_json .= $char;
 188          break;
 189      }
 190    }
 191    return $new_json;
 192  }

 194  /**
 195   * Joins key:value pairs by inner_glue and each pair together by outer_glue
 196   *
 197   * @param string $inner_glue The HTTP method (GET, POST, PUT, DELETE)
 198   * @param string $outer_glue Full URL of the resource to access
 199   * @param array $array Associative array of query parameters
 200   * @return string Urlencoded string of query parameters

 202  function implode_assoc($inner_glue, $outer_glue, $array) {
 203    $output = array();
 204    foreach($array as $key => $item) {
 205      $output[] = $key . $inner_glue . urlencode($item);
 206    }
 207    return implode($outer_glue, $output);
 208  }
