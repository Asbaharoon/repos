 190    public static function from_request($http_method=NULL, $http_url=NULL, $parameters=NULL) {/*{{{*/
 191      $scheme = (!isset($_SERVER['HTTPS']) || $_SERVER['HTTPS'] != "on") ? 'http' : 'https';
 192      @$http_url or $http_url = $scheme . '://' . $_SERVER['HTTP_HOST'] . $_SERVERSERVER['REQUEST_URI'];
 193      @$http_method or $http_method = $_SERVER['REQUEST_METHOD'];
 194

 196  
 197      // let the library user override things however they'd like, if they know
 198      // which parameters to use then go for it, for example XMLRPC might want to
 199      // do this
 200      if ($parameters) {
 201        $req = new OAuthRequest($http_method, $http_url, $parameters);
 202      }
 203      // next check for the auth header, we need to do some extra stuff
 204      // if that is the case, namely suck in the parameters from GET or POST
 205      // so that we can include them in the signature
 206      else if (@substr($request_headers['Authorization'], 0, 5) == "OAuth") {
 207        $header_parameters = OAuthRequest::split_header($request_headers['Authorization']);

 209          $req_parameters = $_GET;

 215        $req = new OAuthRequest($http_method, $http_url, $parameters);
 216      }
 217      else if ($http_method == "GET") {
 218        $req = new OAuthRequest($http_method, $http_url, $_GET);

 255    /**
 256     * Returns the normalized parameters of the request
 257     * 
 258     * This will be all (except oauth_signature) parameters,
 259     * sorted first by key, and if duplicate keys, then by
 260     * value.
 261     *
 262     * The returned string will be all the key=value pairs
 263     * concated by &.
 264     * 
 265     * @return string
 266     */
 267    public function get_signable_parameters() {/*{{{*/
 268      // Grab all parameters
 269      $params = $this->parameters;
 270          
 271      // Remove oauth_signature if present
 272      if (isset($params['oauth_signature'])) {
 273        unset($params['oauth_signature']);
 274      }
 275          
 276      // Urlencode both keys and values
 277      $keys = array_map(array('OAuthUtil', 'urlencodeRFC3986'), array_keys($params));
 278      $values = array_map(array('OAuthUtil', 'urlencodeRFC3986'), array_values($params));
 279      $params = array_combine($keys, $values);
 280  
 281      // Sort by keys (natsort)
 282      uksort($params, 'strnatcmp');
 283  
 284      // Generate key=value pairs
 285      $pairs = array();
 286      foreach ($params as $key=>$value ) {
 287        if (is_array($value)) {
 288          // If the value is an array, it's because there are multiple 
 289          // with the same key, sort them, then add all the pairs
 290          natsort($value);
 291          foreach ($value as $v2) {
 292            $pairs[] = $key . '=' . $v2;
 293          }
 294        } else {
 295          $pairs[] = $key . '=' . $value;
 296        }
 297      }
 298          
 299      // Return the pairs, concated with &
 300      return implode('&', $pairs);
 301    }/*}}}*/

 420    /**
 421     * util function for turning the Authorization: header into
 422     * parameters, has to do some unescaping
 423     */
 424    private static function split_header($header) {/*{{{*/
 425      // this should be a regex
 426      // error cases: commas in parameter values
 427      $parts = explode(",", $header);
 428      $out = array();
 429      foreach ($parts as $param) {
 430        $param = ltrim($param);
 431        // skip the "realm" param, nobody ever uses it anyway
 432        if (substr($param, 0, 5) != "oauth") continue;
 433  
 434        $param_parts = explode("=", $param);
 435  
 436        // rawurldecode() used because urldecode() will turn a "+" in the
 437        // value into a space
 438        $out[$param_parts[0]] = rawurldecode(substr($param_parts[1], 1, -1));
 439      }
 440      return $out;
 441    }/*}}}*/

 443    /**
 444     * helper to try to sort out headers for people who aren't running apache
 445     */
 446    private static function get_headers() {/*{{{*/
 447      if (function_exists('apache_request_headers')) {
 448        // we need this to get the actual Authorization: header
 449        // because apache tends to tell us it doesn't exist
 450        return apache_request_headers();
 451      }
 452      // otherwise we don't have apache and are just going to have to hope
 453      // that $_SERVER actually contains what we need
 454      $out = array();
 455      foreach ($_SERVER as $key => $value) {
 456        if (substr($key, 0, 5) == "HTTP_") {
 457          // this is chaos, basically it is just there to capitalize the first
 458          // letter of every word that is not an initial HTTP and strip HTTP
 459          // code from przemek
 460          $key = str_replace(" ", "-", ucwords(strtolower(str_replace("_", " ", substr($key, 5)))));
 461          $out[$key] = $value;
 462        }
 463      }
 464      return $out;
 465    }/*}}}*/
