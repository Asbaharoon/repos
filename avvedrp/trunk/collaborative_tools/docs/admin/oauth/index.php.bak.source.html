 193  /**
 194   * Upgrades an OAuth request token to an access token.
 195   *
 196   * @param string $request_token_str An authorized OAuth request token
 197   * @return string The access token
 198   */
 199  function getAccessToken($request_token_str) {
 200    global $consumer, $sig_method;
 201  
 202    $token = new OAuthToken($request_token_str, NULL);
 203  
 204    $token_endpoint = 'https://www.google.com/accounts/OAuthGetAccessToken';
 205    $request = OAuthRequest::from_consumer_and_token($consumer, $token, 'GET',
 206                                                     $token_endpoint);
 207    $request->sign_request($sig_method, $consumer, $token);
 208  
 209    $response = send_signed_request($request->get_normalized_http_method(),
 210                                    $token_endpoint, $request->to_header(), NULL,
 211                                    false);
 212  
 213    // Parse out oauth_token (access token) and oauth_token_secret
 214    preg_match('/oauth_token=(.*)&oauth_token_secret=(.*)/', $response, $matches);
 215    $access_token = new OAuthToken(urldecode($matches[1]),
 216                                   urldecode($matches[2]));
 217  
 218    return $access_token;
 219  }

 221  /**
 222   * Creates an HTML list of each <entry>'s title.
 223   *
 224   * @param Zend_Gdata_Feed $feed A Gdata feed object
 225   * @return string The HTML of entries
 226   */
 227  function listEntries($feed) {
 228    $str = '<ul>';
 229    foreach($feed->entries as $entry) {
 230      // Find the URL of the HTML view of the document.
 231      foreach ($entry->link as $link) {
 232        if ($link->getRel() === 'alternate') {
 233          $alternateLink = $link->getHref();
 234        }
 235      }
 236      $str .= "<li><a href=\"{$alternateLink}\" target=\"new\">{$entry->title->text}</a></li>";
 237    }
 238    $str .= '</ul>';
 239  
 240    return $str;
 241  }
