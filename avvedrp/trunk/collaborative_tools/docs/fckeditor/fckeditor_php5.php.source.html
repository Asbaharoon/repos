  39  function FCKeditor_IsCompatibleBrowser()
  40  {
  41      if ( isset( $_SERVER ) ) {
  42          $sAgent = $_SERVER['HTTP_USER_AGENT'] ;
  43      }
  44      else {
  45          global $HTTP_SERVER_VARS ;
  46          if ( isset( $HTTP_SERVER_VARS ) ) {
  47              $sAgent = $HTTP_SERVER_VARS['HTTP_USER_AGENT'] ;
  48          }
  49          else {
  50              global $HTTP_USER_AGENT ;
  51              $sAgent = $HTTP_USER_AGENT ;
  52          }
  53      }
  54  
  55      if ( strpos($sAgent, 'MSIE') !== false && strpos($sAgent, 'mac') === false && strpos($sAgent, 'Opera') === false )
  56      {
  57          $iVersion = (float)substr($sAgent, strpos($sAgent, 'MSIE') + 5, 3) ;
  58          return ($iVersion >= 5.5) ;
  59      }
  60      else if ( strpos($sAgent, 'Gecko/') !== false )
  61      {
  62          $iVersion = (int)substr($sAgent, strpos($sAgent, 'Gecko/') + 6, 8) ;
  63          return ($iVersion >= 20030210) ;
  64      }
  65      else if ( strpos($sAgent, 'Opera/') !== false )
  66      {
  67          $fVersion = (float)substr($sAgent, strpos($sAgent, 'Opera/') + 6, 4) ;
  68          return ($fVersion >= 9.5) ;
  69      }
  70      else if ( preg_match( "|AppleWebKit/(\d+)|i", $sAgent, $matches ) )
  71      {
  72          $iVersion = $matches[1] ;
  73          return ( $matches[1] >= 522 ) ;
  74      }
  75      else
  76          return false ;
  77  }

 164      /**
 165       * Return the HTML code required to run FCKeditor.
 166       *
 167       * @return string
 168       */
 169  	public function CreateHtml()
 170      {
 171          $HtmlValue = htmlspecialchars( $this->Value ) ;
 172  
 173          $Html = '' ;
 174  
 175          if ( $this->IsCompatible() )
 176          {
 177              if ( isset( $_GET['fcksource'] ) && $_GET['fcksource'] == "true" )
 178                  $File = 'fckeditor.original.html' ;
 179              else
 180                  $File = 'fckeditor.html' ;
 181  
 182              $Link = "{$this->BasePath}editor/{$File}?InstanceName={$this->InstanceName}" ;
 183  
 184              if ( $this->ToolbarSet != '' )
 185                  $Link .= "&amp;Toolbar={$this->ToolbarSet}" ;
 186  
 187              // Render the linked hidden field.
 188              $Html .= "<input type=\"hidden\" id=\"{$this->InstanceName}\" name=\"{$this->InstanceName}\" value=\"{$HtmlValue}\" style=\"display:none\" />" ;
 189  
 190              // Render the configurations hidden field.
 191              $Html .= "<input type=\"hidden\" id=\"{$this->InstanceName}___Config\" value=\"" . $this->GetConfigFieldString() . "\" style=\"display:none\" />" ;
 192  
 193              // Render the editor IFRAME.
 194              $Html .= "<iframe id=\"{$this->InstanceName}___Frame\" src=\"{$Link}\" width=\"{$this->Width}\" height=\"{$this->Height}\" frameborder=\"0\" scrolling=\"no\"></iframe>" ;
 195          }
 196          else
 197          {
 198              if ( strpos( $this->Width, '%' ) === false )
 199                  $WidthCSS = $this->Width . 'px' ;
 200              else
 201                  $WidthCSS = $this->Width ;
 202  
 203              if ( strpos( $this->Height, '%' ) === false )
 204                  $HeightCSS = $this->Height . 'px' ;
 205              else
 206                  $HeightCSS = $this->Height ;
 207  
 208              $Html .= "<textarea name=\"{$this->InstanceName}\" rows=\"4\" cols=\"40\" style=\"width: {$WidthCSS}; height: {$HeightCSS}\">{$HtmlValue}</textarea>" ;
 209          }
 210  
 211          return $Html ;
 212      }

 224      /**
 225       * Get settings from Config array as a single string.
 226       *
 227       * @access protected
 228       * @return string
 229       */
 230  	public function GetConfigFieldString()
 231      {
 232          $sParams = '' ;
 233          $bFirst = true ;
 234  
 235          foreach ( $this->Config as $sKey => $sValue )
 236          {
 237              if ( $bFirst == false )
 238                  $sParams .= '&amp;' ;
 239              else
 240                  $bFirst = false ;
 241  
 242              if ( $sValue === true )
 243                  $sParams .= $this->EncodeConfig( $sKey ) . '=true' ;
 244              else if ( $sValue === false )
 245                  $sParams .= $this->EncodeConfig( $sKey ) . '=false' ;
 246              else
 247                  $sParams .= $this->EncodeConfig( $sKey ) . '=' . $this->EncodeConfig( $sValue ) ;
 248          }
 249  
 250          return $sParams ;
 251      }

 253      /**
 254       * Encode characters that may break the configuration string
 255       * generated by GetConfigFieldString().
 256       *
 257       * @access protected
 258       * @param string $valueToEncode
 259       * @return string
 260       */
 261  	public function EncodeConfig( $valueToEncode )
 262      {
 263          $chars = array(
 264              '&' => '%26',
 265              '=' => '%3D',
 266              '"' => '%22' ) ;
 267  
 268          return strtr( $valueToEncode,  $chars ) ;
 269      }
