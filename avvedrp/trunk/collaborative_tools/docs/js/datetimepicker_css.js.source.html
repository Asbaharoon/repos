  85  function NewCssCal(pCtrl,pFormat,pScroller,pShowTime,pTimeMode,pHideSeconds) {
  86      // get current date and time
  87  
  88      dtToday = new Date();
  89      Cal=new Calendar(dtToday);
  90  
  91      if ((pShowTime!=null) && (pShowTime)) {
  92          Cal.ShowTime=true;
  93          if ((pTimeMode!=null) &&((pTimeMode=='12')||(pTimeMode=='24')))    {
  94              TimeMode=pTimeMode;
  95          }
  96  
  97          else TimeMode='24';
  98  
  99          if (pHideSeconds!=null)
 100          {
 101              if (pHideSeconds)
 102              {Cal.ShowSeconds=false;}
 103              else
 104              {Cal.ShowSeconds=true;}
 105  
 106          }
 107  
 108          else
 109          {
 110              Cal.ShowSeconds=false;
 111          }    
 112  
 113      }
 114  
 115      if (pCtrl!=null)
 116  
 117          Cal.Ctrl=pCtrl;
 118  
 119  
 120      if (pFormat!=null)
 121          Cal.Format=pFormat.toUpperCase();
 122      else 
 123          Cal.Format="MMDDYYYY";
 124  
 125      if (pScroller!=null) {
 126          if (pScroller.toUpperCase()=="ARROW") {
 127              Cal.Scroller="ARROW";
 128          }
 129          else {
 130              Cal.Scroller="DROPDOWN";
 131          }
 132      }
 133  
 134      exDateTime=document.getElementById(pCtrl).value;
 135  
 136      if (exDateTime!="")    { //Parse existing Date String
 137          var Sp1;//Index of Date Separator 1
 138          var Sp2;//Index of Date Separator 2 
 139          var tSp1;//Index of Time Separator 1
 140          var tSp1;//Index of Time Separator 2
 141          var strMonth;
 142          var strDate;
 143          var strYear;
 144          var intMonth;
 145          var YearPattern;
 146          var strHour;
 147          var strMinute;
 148          var strSecond;
 149          var winHeight;
 150          //parse month
 151  
 152          Sp1=exDateTime.indexOf(DateSeparator,0)
 153          Sp2=exDateTime.indexOf(DateSeparator,(parseInt(Sp1)+1));
 154          var offset=parseInt(Cal.Format.toUpperCase().lastIndexOf("M"))-parseInt(Cal.Format.toUpperCase().indexOf("M"))-1;
 155  
 156          if ((Cal.Format.toUpperCase()=="DDMMYYYY") || (Cal.Format.toUpperCase()=="DDMMMYYYY")) {
 157              if (DateSeparator=="") {
 158                  strMonth=exDateTime.substring(2,4+offset);
 159                  strDate=exDateTime.substring(0,2);
 160                  strYear=exDateTime.substring(4+offset,8+offset);
 161              }
 162              else {
 163                  if(exDateTime.indexOf("D*") != -1) {   //DTG
 164                      strMonth = exDateTime.substring(8, 11);
 165                      strDate  = exDateTime.substring(0, 2);
 166                      strYear  = "20" + exDateTime.substring(11, 13);  //Hack, nur für Jahreszahlen ab 2000
 167                      
 168                  } else {
 169                      strMonth=exDateTime.substring(Sp1+1,Sp2);
 170                      strDate=exDateTime.substring(0,Sp1);
 171                      strYear=exDateTime.substring(Sp2+1,Sp2+5);
 172                  }
 173              }
 174  
 175          }
 176  
 177          else if ((Cal.Format.toUpperCase()=="MMDDYYYY") || (Cal.Format.toUpperCase()=="MMMDDYYYY")) {
 178  
 179              if (DateSeparator=="") {
 180                  strMonth=exDateTime.substring(0,2+offset);
 181                  strDate=exDateTime.substring(2+offset,4+offset);
 182                  strYear=exDateTime.substring(4+offset,8+offset);
 183              }
 184  
 185              else {
 186  
 187                  strMonth=exDateTime.substring(0,Sp1);
 188                  strDate=exDateTime.substring(Sp1+1,Sp2);
 189                  strYear=exDateTime.substring(Sp2+1,Sp2+5);
 190              }
 191  
 192          }
 193  
 194          else if ((Cal.Format.toUpperCase()=="YYYYMMDD") || (Cal.Format.toUpperCase()=="YYYYMMMDD")) {
 195  
 196              if (DateSeparator=="") {
 197                  strMonth=exDateTime.substring(4,6+offset);
 198                  strDate=exDateTime.substring(6+offset,8+offset);
 199                  strYear=exDateTime.substring(0,4);
 200              }
 201  
 202              else {
 203                  strMonth=exDateTime.substring(Sp1+1,Sp2);
 204                  strDate=exDateTime.substring(Sp2+1,Sp2+3);
 205                  strYear=exDateTime.substring(0,Sp1);
 206              }
 207  
 208          }
 209  
 210          else if ((Cal.Format.toUpperCase()=="YYMMDD") || (Cal.Format.toUpperCase()=="YYMMMDD")) {
 211  
 212              if (DateSeparator=="") {
 213                  strMonth=exDateTime.substring(2,4+offset);
 214                  strDate=exDateTime.substring(4+offset,6+offset);
 215                  strYear=exDateTime.substring(0,2);
 216              }
 217  
 218              else {
 219                  strMonth=exDateTime.substring(Sp1+1,Sp2);
 220                  strDate=exDateTime.substring(Sp2+1,Sp2+3);
 221                  strYear=exDateTime.substring(0,Sp1);
 222              }
 223  
 224          }        
 225  
 226          if (isNaN(strMonth))
 227              intMonth=Cal.GetMonthIndex(strMonth);
 228          else
 229              intMonth=parseInt(strMonth,10)-1;    
 230  
 231          if ((parseInt(intMonth,10)>=0) && (parseInt(intMonth,10)<12))
 232              Cal.Month=intMonth;
 233  
 234          //end parse month
 235  
 236          //parse Date
 237  
 238          if ((parseInt(strDate,10)<=Cal.GetMonDays()) && (parseInt(strDate,10)>=1))
 239              Cal.Date=strDate;
 240          //end parse Date
 241  
 242          //parse year
 243  
 244          YearPattern=/^\d{4}$/;
 245          if (YearPattern.test(strYear))
 246              Cal.Year=parseInt(strYear,10);
 247  
 248          //end parse year
 249  
 250          //parse time
 251  
 252          if (Cal.ShowTime==true)    {
 253  
 254              //parse AM or PM
 255  
 256              if (TimeMode==12) {
 257                  strAMPM=exDateTime.substring(exDateTime.length-2,exDateTime.length)
 258                  Cal.AMorPM=strAMPM;
 259  
 260              }
 261  
 262              tSp1=exDateTime.indexOf(":",0)
 263              tSp2=exDateTime.indexOf(":",(parseInt(tSp1)+1));
 264              if (tSp1>0)    {
 265  
 266                  strHour=exDateTime.substring(tSp1,(tSp1)-2);
 267                  Cal.SetHour(strHour);
 268  
 269                  strMinute=exDateTime.substring(tSp1+1,tSp1+3);
 270                  Cal.SetMinute(strMinute);
 271  
 272                  strSecond=exDateTime.substring(tSp2+1,tSp2+3);
 273                  Cal.SetSecond(strSecond);
 274  
 275              } else if(exDateTime.indexOf("D*") != -1) {   //DTG
 276                  strHour = exDateTime.substring(2, 4);
 277                  Cal.SetHour(strHour);
 278                  strMinute = exDateTime.substring(4, 6);
 279                  Cal.SetMinute(strMinute);
 280                  
 281              }
 282          }    
 283  
 284      }
 285  
 286      selDate=new Date(Cal.Year,Cal.Month,Cal.Date);//version 1.7
 287      RenderCssCal(true);
 288  }
