 259      /**
 260       * Set the URI for the next request
 261       *
 262       * @param  Zend_Uri_Http|string $uri
 263       * @return Zend_Http_Client
 264       * @throws Zend_Http_Client_Exception
 265       */
 266      public function setUri($uri)
 267      {
 268          if (is_string($uri)) {
 269              $uri = Zend_Uri::factory($uri);
 270          }
 271  
 272          if (!$uri instanceof Zend_Uri_Http) {
 273              /** @see Zend_Http_Client_Exception */
 274              require_once 'Zend/Http/Client/Exception.php';
 275              throw new Zend_Http_Client_Exception('Passed parameter is not a valid HTTP URI.');
 276          }
 277  
 278          // Set auth if username and password has been specified in the uri
 279          if ($uri->getUsername() && $uri->getPassword()) {
 280              $this->setAuth($uri->getUsername(), $uri->getPassword());
 281          }
 282  
 283          // We have no ports, set the defaults
 284          if (! $uri->getPort()) {
 285              $uri->setPort(($uri->getScheme() == 'https' ? 443 : 80));
 286          }
 287  
 288          $this->uri = $uri;
 289  
 290          return $this;
 291      }

 366      /**
 367       * Set one or more request headers
 368       *
 369       * This function can be used in several ways to set the client's request
 370       * headers:
 371       * 1. By providing two parameters: $name as the header to set (eg. 'Host')
 372       *    and $value as it's value (eg. 'www.example.com').
 373       * 2. By providing a single header string as the only parameter
 374       *    eg. 'Host: www.example.com'
 375       * 3. By providing an array of headers as the first parameter
 376       *    eg. array('host' => 'www.example.com', 'x-foo: bar'). In This case
 377       *    the function will call itself recursively for each array item.
 378       *
 379       * @param string|array $name Header name, full header string ('Header: value')
 380       *     or an array of headers
 381       * @param mixed $value Header value or null
 382       * @return Zend_Http_Client
 383       * @throws Zend_Http_Client_Exception
 384       */
 385      public function setHeaders($name, $value = null)
 386      {
 387          // If we got an array, go recusive!
 388          if (is_array($name)) {
 389              foreach ($name as $k => $v) {
 390                  if (is_string($k)) {
 391                      $this->setHeaders($k, $v);
 392                  } else {
 393                      $this->setHeaders($v, null);
 394                  }
 395              }
 396          } else {
 397              // Check if $name needs to be split
 398              if ($value === null && (strpos($name, ':') > 0)) {
 399                  list($name, $value) = explode(':', $name, 2);
 400              }
 401  
 402              // Make sure the name is valid if we are in strict mode
 403              if ($this->config['strict'] && (! preg_match('/^[a-zA-Z0-9-]+$/', $name))) {
 404                  /** @see Zend_Http_Client_Exception */
 405                  require_once 'Zend/Http/Client/Exception.php';
 406                  throw new Zend_Http_Client_Exception("{$name} is not a valid HTTP header name");
 407              }
 408  
 409              $normalized_name = strtolower($name);
 410  
 411              // If $value is null or false, unset the header
 412              if ($value === null || $value === false) {
 413                  unset($this->headers[$normalized_name]);
 414  
 415              // Else, set the header
 416              } else {
 417                  // Header names are stored lowercase internally.
 418                  if (is_string($value)) {
 419                      $value = trim($value);
 420                  }
 421                  $this->headers[$normalized_name] = array($name, $value);
 422              }
 423          }
 424  
 425          return $this;
 426      }

 488       * @param string $type GET or POST
 489       * @param string $name
 490       * @param string $value
 491       * @return null
 492       */
 493      protected function _setParameter($type, $name, $value)
 494      {
 495          $parray = array();
 496          $type = strtolower($type);
 497          switch ($type) {
 498              case 'get':
 499                  $parray = &$this->paramsGet;
 500                  break;
 501              case 'post':
 502                  $parray = &$this->paramsPost;
 503                  break;
 504          }
 505  
 506          if ($value === null) {
 507              if (isset($parray[$name])) unset($parray[$name]);
 508          } else {
 509              $parray[$name] = $value;
 510          }
 511      }

 523      /**
 524       * Set HTTP authentication parameters
 525       *
 526       * $type should be one of the supported types - see the self::AUTH_*
 527       * constants.
 528       *
 529       * To enable authentication:
 530       * <code>
 531       * $this->setAuth('shahar', 'secret', Zend_Http_Client::AUTH_BASIC);
 532       * </code>
 533       *
 534       * To disable authentication:
 535       * <code>
 536       * $this->setAuth(false);
 537       * </code>
 538       *
 539       * @see http://www.faqs.org/rfcs/rfc2617.html
 540       * @param string|false $user User name or false disable authentication
 541       * @param string $password Password
 542       * @param string $type Authentication type
 543       * @return Zend_Http_Client
 544       * @throws Zend_Http_Client_Exception
 545       */
 546      public function setAuth($user, $password = '', $type = self::AUTH_BASIC)
 547      {
 548          // If we got false or null, disable authentication
 549          if ($user === false || $user === null) {
 550              $this->auth = null;
 551  
 552              // Clear the auth information in the uri instance as well
 553              if ($this->uri instanceof Zend_Uri_Http) {
 554                  $this->getUri()->setUsername('');
 555                  $this->getUri()->setPassword('');
 556              }
 557          // Else, set up authentication
 558          } else {
 559              // Check we got a proper authentication type
 560              if (! defined('self::AUTH_' . strtoupper($type))) {
 561                  /** @see Zend_Http_Client_Exception */
 562                  require_once 'Zend/Http/Client/Exception.php';
 563                  throw new Zend_Http_Client_Exception("Invalid or not supported authentication type: '$type'");
 564              }
 565  
 566              $this->auth = array(
 567                  'user' => (string) $user,
 568                  'password' => (string) $password,
 569                  'type' => $type
 570              );
 571          }
 572  
 573          return $this;
 574      }

 617      /**
 618       * Add a cookie to the request. If the client has no Cookie Jar, the cookies
 619       * will be added directly to the headers array as "Cookie" headers.
 620       *
 621       * @param Zend_Http_Cookie|string $cookie
 622       * @param string|null $value If "cookie" is a string, this is the cookie value.
 623       * @return Zend_Http_Client
 624       * @throws Zend_Http_Client_Exception
 625       */
 626      public function setCookie($cookie, $value = null)
 627      {
 628          if (! class_exists('Zend_Http_Cookie')) {
 629              require_once 'Zend/Http/Cookie.php';
 630          }
 631  
 632          if (is_array($cookie)) {
 633              foreach ($cookie as $c => $v) {
 634                  if (is_string($c)) {
 635                      $this->setCookie($c, $v);
 636                  } else {
 637                      $this->setCookie($v);
 638                  }
 639              }
 640  
 641              return $this;
 642          }
 643  
 644          if ($value !== null && $this->config['encodecookies']) {
 645              $value = urlencode($value);
 646          }
 647  
 648          if (isset($this->cookiejar)) {
 649              if ($cookie instanceof Zend_Http_Cookie) {
 650                  $this->cookiejar->addCookie($cookie);
 651              } elseif (is_string($cookie) && $value !== null) {
 652                  $cookie = Zend_Http_Cookie::fromString("{$cookie}={$value}",
 653                                                         $this->uri,
 654                                                         $this->config['encodecookies']);
 655                  $this->cookiejar->addCookie($cookie);
 656              }
 657          } else {
 658              if ($cookie instanceof Zend_Http_Cookie) {
 659                  $name = $cookie->getName();
 660                  $value = $cookie->getValue();
 661                  $cookie = $name;
 662              }
 663  
 664              if (preg_match("/[=,; \t\r\n\013\014]/", $cookie)) {
 665                  /** @see Zend_Http_Client_Exception */
 666                  require_once 'Zend/Http/Client/Exception.php';
 667                  throw new Zend_Http_Client_Exception("Cookie name cannot contain these characters: =,; \t\r\n\013\014 ({$cookie})");
 668              }
 669  
 670              $value = addslashes($value);
 671  
 672              if (! isset($this->headers['cookie'])) {
 673                  $this->headers['cookie'] = array('Cookie', '');
 674              }
 675              $this->headers['cookie'][1] .= $cookie . '=' . $value . '; ';
 676          }
 677  
 678          return $this;
 679      }

 681      /**
 682       * Set a file to upload (using a POST request)
 683       *
 684       * Can be used in two ways:
 685       *
 686       * 1. $data is null (default): $filename is treated as the name if a local file which
 687       *    will be read and sent. Will try to guess the content type using mime_content_type().
 688       * 2. $data is set - $filename is sent as the file name, but $data is sent as the file
 689       *    contents and no file is read from the file system. In this case, you need to
 690       *    manually set the Content-Type ($ctype) or it will default to
 691       *    application/octet-stream.
 692       *
 693       * @param string $filename Name of file to upload, or name to save as
 694       * @param string $formname Name of form element to send as
 695       * @param string $data Data to send (if null, $filename is read and sent)
 696       * @param string $ctype Content type to use (if $data is set and $ctype is
 697       *     null, will be application/octet-stream)
 698       * @return Zend_Http_Client
 699       * @throws Zend_Http_Client_Exception
 700       */
 701      public function setFileUpload($filename, $formname, $data = null, $ctype = null)
 702      {
 703          if ($data === null) {
 704              if (($data = @file_get_contents($filename)) === false) {
 705                  /** @see Zend_Http_Client_Exception */
 706                  require_once 'Zend/Http/Client/Exception.php';
 707                  throw new Zend_Http_Client_Exception("Unable to read file '{$filename}' for upload");
 708              }
 709  
 710              if (! $ctype) {
 711                  $ctype = $this->_detectFileMimeType($filename);
 712              }
 713          }
 714  
 715          // Force enctype to multipart/form-data
 716          $this->setEncType(self::ENC_FORMDATA);
 717  
 718          $this->files[] = array(
 719              'formname' => $formname,
 720              'filename' => basename($filename),
 721              'ctype'    => $ctype,
 722              'data'     => $data
 723          );
 724  
 725          return $this;
 726      }

 745       * 1. For advanced user who would like to set their own data, already encoded
 746       * 2. For backwards compatibilty: If someone uses the old post($data) method.
 747       *    this method will be used to set the encoded data.
 748       *
 749       * $data can also be stream (such as file) from which the data will be read.
 750       *
 751       * @param string|resource $data
 752       * @param string $enctype
 753       * @return Zend_Http_Client
 754       */
 755      public function setRawData($data, $enctype = null)
 756      {
 757          $this->raw_post_data = $data;
 758          $this->setEncType($enctype);
 759          if (is_resource($data)) {
 760              // We've got stream data
 761              $stat = @fstat($data);
 762              if($stat) {
 763                  $this->setHeaders(self::CONTENT_LENGTH, $stat['size']);
 764              }
 765          }
 766          return $this;
 767      }

 923      /**
 924       * Send the HTTP request and return an HTTP response object
 925       *
 926       * @param string $method
 927       * @return Zend_Http_Response
 928       * @throws Zend_Http_Client_Exception
 929       */
 930      public function request($method = null)
 931      {
 932          if (! $this->uri instanceof Zend_Uri_Http) {
 933              /** @see Zend_Http_Client_Exception */
 934              require_once 'Zend/Http/Client/Exception.php';
 935              throw new Zend_Http_Client_Exception('No valid URI has been passed to the client');
 936          }
 937  
 938          if ($method) {
 939              $this->setMethod($method);
 940          }
 941          $this->redirectCounter = 0;
 942          $response = null;
 943  
 944          // Make sure the adapter is loaded
 945          if ($this->adapter == null) {
 946              $this->setAdapter($this->config['adapter']);
 947          }
 948  
 949          // Send the first request. If redirected, continue.
 950          do {
 951              // Clone the URI and add the additional GET parameters to it
 952              $uri = clone $this->uri;
 953              if (! empty($this->paramsGet)) {
 954                  $query = $uri->getQuery();
 955                     if (! empty($query)) {
 956                         $query .= '&';
 957                     }
 958                  $query .= http_build_query($this->paramsGet, null, '&');
 959  
 960                  $uri->setQuery($query);
 961              }
 962  
 963              $body = $this->_prepareBody();
 964              $headers = $this->_prepareHeaders();
 965  
 966              // check that adapter supports streaming before using it
 967              if(is_resource($body) && !($this->adapter instanceof Zend_Http_Client_Adapter_Stream)) {
 968                  /** @see Zend_Http_Client_Exception */
 969                  require_once 'Zend/Http/Client/Exception.php';
 970                  throw new Zend_Http_Client_Exception('Adapter does not support streaming');
 971              }
 972  
 973              // Open the connection, send the request and read the response
 974              $this->adapter->connect($uri->getHost(), $uri->getPort(),
 975                  ($uri->getScheme() == 'https' ? true : false));
 976  
 977              if($this->config['output_stream']) {
 978                  if($this->adapter instanceof Zend_Http_Client_Adapter_Stream) {
 979                      $stream = $this->_openTempStream();
 980                      $this->adapter->setOutputStream($stream);
 981                  } else {
 982                      /** @see Zend_Http_Client_Exception */
 983                      require_once 'Zend/Http/Client/Exception.php';
 984                      throw new Zend_Http_Client_Exception('Adapter does not support streaming');
 985                  }
 986              }
 987  
 988              $this->last_request = $this->adapter->write($this->method,
 989                  $uri, $this->config['httpversion'], $headers, $body);
 990  
 991              $response = $this->adapter->read();
 992              if (! $response) {
 993                  /** @see Zend_Http_Client_Exception */
 994                  require_once 'Zend/Http/Client/Exception.php';
 995                  throw new Zend_Http_Client_Exception('Unable to read response, or response is empty');
 996              }
 997  
 998              if($this->config['output_stream']) {
 999                  rewind($stream);
1000                  // cleanup the adapter
1001                  $this->adapter->setOutputStream(null);
1002                  $response = Zend_Http_Response_Stream::fromStream($response, $stream);
1003                  $response->setStreamName($this->_stream_name);
1004                  if(!is_string($this->config['output_stream'])) {
1005                      // we used temp name, will need to clean up
1006                      $response->setCleanup(true);
1007                  }
1008              } else {
1009                  $response = Zend_Http_Response::fromString($response);
1010              }
1011  
1012              if ($this->config['storeresponse']) {
1013                  $this->last_response = $response;
1014              }
1015  
1016              // Load cookies into cookie jar
1017              if (isset($this->cookiejar)) {
1018                  $this->cookiejar->addCookiesFromResponse($response, $uri);
1019              }
1020  
1021              // If we got redirected, look for the Location header
1022              if ($response->isRedirect() && ($location = $response->getHeader('location'))) {
1023  
1024                  // Check whether we send the exact same request again, or drop the parameters
1025                  // and send a GET request
1026                  if ($response->getStatus() == 303 ||
1027                     ((! $this->config['strictredirects']) && ($response->getStatus() == 302 ||
1028                         $response->getStatus() == 301))) {
1029  
1030                      $this->resetParameters();
1031                      $this->setMethod(self::GET);
1032                  }
1033  
1034                  // If we got a well formed absolute URI
1035                  if (Zend_Uri_Http::check($location)) {
1036                      $this->setHeaders('host', null);
1037                      $this->setUri($location);
1038  
1039                  } else {
1040  
1041                      // Split into path and query and set the query
1042                      if (strpos($location, '?') !== false) {
1043                          list($location, $query) = explode('?', $location, 2);
1044                      } else {
1045                          $query = '';
1046                      }
1047                      $this->uri->setQuery($query);
1048  
1049                      // Else, if we got just an absolute path, set it
1050                      if(strpos($location, '/') === 0) {
1051                          $this->uri->setPath($location);
1052  
1053                          // Else, assume we have a relative path
1054                      } else {
1055                          // Get the current path directory, removing any trailing slashes
1056                          $path = $this->uri->getPath();
1057                          $path = rtrim(substr($path, 0, strrpos($path, '/')), "/");
1058                          $this->uri->setPath($path . '/' . $location);
1059                      }
1060                  }
1061                  ++$this->redirectCounter;
1062  
1063              } else {
1064                  // If we didn't get any location, stop redirecting
1065                  break;
1066              }
1067  
1068          } while ($this->redirectCounter < $this->config['maxredirects']);
1069  
1070          return $response;
1071      }

1073      /**
1074       * Prepare the request headers
1075       *
1076       * @return array
1077       */
1078      protected function _prepareHeaders()
1079      {
1080          $headers = array();
1081  
1082          // Set the host header
1083          if (! isset($this->headers['host'])) {
1084              $host = $this->uri->getHost();
1085  
1086              // If the port is not default, add it
1087              if (! (($this->uri->getScheme() == 'http' && $this->uri->getPort() == 80) ||
1088                    ($this->uri->getScheme() == 'https' && $this->uri->getPort() == 443))) {
1089                  $host .= ':' . $this->uri->getPort();
1090              }
1091  
1092              $headers[] = "Host: {$host}";
1093          }
1094  
1095          // Set the connection header
1096          if (! isset($this->headers['connection'])) {
1097              if (! $this->config['keepalive']) {
1098                  $headers[] = "Connection: close";
1099              }
1100          }
1101  
1102          // Set the Accept-encoding header if not set - depending on whether
1103          // zlib is available or not.
1104          if (! isset($this->headers['accept-encoding'])) {
1105              if (function_exists('gzinflate')) {
1106                  $headers[] = 'Accept-encoding: gzip, deflate';
1107              } else {
1108                  $headers[] = 'Accept-encoding: identity';
1109              }
1110          }
1111  
1112          // Set the Content-Type header
1113          if ($this->method == self::POST &&
1114             (! isset($this->headers[strtolower(self::CONTENT_TYPE)]) && isset($this->enctype))) {
1115  
1116              $headers[] = self::CONTENT_TYPE . ': ' . $this->enctype;
1117          }
1118  
1119          // Set the user agent header
1120          if (! isset($this->headers['user-agent']) && isset($this->config['useragent'])) {
1121              $headers[] = "User-Agent: {$this->config['useragent']}";
1122          }
1123  
1124          // Set HTTP authentication if needed
1125          if (is_array($this->auth)) {
1126              $auth = self::encodeAuthHeader($this->auth['user'], $this->auth['password'], $this->auth['type']);
1127              $headers[] = "Authorization: {$auth}";
1128          }
1129  
1130          // Load cookies from cookie jar
1131          if (isset($this->cookiejar)) {
1132              $cookstr = $this->cookiejar->getMatchingCookies($this->uri,
1133                  true, Zend_Http_CookieJar::COOKIE_STRING_CONCAT);
1134  
1135              if ($cookstr) {
1136                  $headers[] = "Cookie: {$cookstr}";
1137              }
1138          }
1139  
1140          // Add all other user defined headers
1141          foreach ($this->headers as $header) {
1142              list($name, $value) = $header;
1143              if (is_array($value)) {
1144                  $value = implode(', ', $value);
1145              }
1146  
1147              $headers[] = "$name: $value";
1148          }
1149  
1150          return $headers;
1151      }

1153      /**
1154       * Prepare the request body (for POST and PUT requests)
1155       *
1156       * @return string
1157       * @throws Zend_Http_Client_Exception
1158       */
1159      protected function _prepareBody()
1160      {
1161          // According to RFC2616, a TRACE request should not have a body.
1162          if ($this->method == self::TRACE) {
1163              return '';
1164          }
1165  
1166          if (isset($this->raw_post_data) && is_resource($this->raw_post_data)) {
1167              return $this->raw_post_data;
1168          }
1169          // If mbstring overloads substr and strlen functions, we have to
1170          // override it's internal encoding
1171          if (function_exists('mb_internal_encoding') &&
1172             ((int) ini_get('mbstring.func_overload')) & 2) {
1173  
1174              $mbIntEnc = mb_internal_encoding();
1175              mb_internal_encoding('ASCII');
1176          }
1177  
1178          // If we have raw_post_data set, just use it as the body.
1179          if (isset($this->raw_post_data)) {
1180              $this->setHeaders(self::CONTENT_LENGTH, strlen($this->raw_post_data));
1181              if (isset($mbIntEnc)) {
1182                  mb_internal_encoding($mbIntEnc);
1183              }
1184  
1185              return $this->raw_post_data;
1186          }
1187  
1188          $body = '';
1189  
1190          // If we have files to upload, force enctype to multipart/form-data
1191          if (count ($this->files) > 0) {
1192              $this->setEncType(self::ENC_FORMDATA);
1193          }
1194  
1195          // If we have POST parameters or files, encode and add them to the body
1196          if (count($this->paramsPost) > 0 || count($this->files) > 0) {
1197              switch($this->enctype) {
1198                  case self::ENC_FORMDATA:
1199                      // Encode body as multipart/form-data
1200                      $boundary = '---ZENDHTTPCLIENT-' . md5(microtime());
1201                      $this->setHeaders(self::CONTENT_TYPE, self::ENC_FORMDATA . "; boundary={$boundary}");
1202  
1203                      // Get POST parameters and encode them
1204                      $params = self::_flattenParametersArray($this->paramsPost);
1205                      foreach ($params as $pp) {
1206                          $body .= self::encodeFormData($boundary, $pp[0], $pp[1]);
1207                      }
1208  
1209                      // Encode files
1210                      foreach ($this->files as $file) {
1211                          $fhead = array(self::CONTENT_TYPE => $file['ctype']);
1212                          $body .= self::encodeFormData($boundary, $file['formname'], $file['data'], $file['filename'], $fhead);
1213                      }
1214  
1215                      $body .= "--{$boundary}--\r\n";
1216                      break;
1217  
1218                  case self::ENC_URLENCODED:
1219                      // Encode body as application/x-www-form-urlencoded
1220                      $this->setHeaders(self::CONTENT_TYPE, self::ENC_URLENCODED);
1221                      $body = http_build_query($this->paramsPost, '', '&');
1222                      break;
1223  
1224                  default:
1225                      if (isset($mbIntEnc)) {
1226                          mb_internal_encoding($mbIntEnc);
1227                      }
1228  
1229                      /** @see Zend_Http_Client_Exception */
1230                      require_once 'Zend/Http/Client/Exception.php';
1231                      throw new Zend_Http_Client_Exception("Cannot handle content type '{$this->enctype}' automatically." .
1232                          " Please use Zend_Http_Client::setRawData to send this kind of content.");
1233                      break;
1234              }
1235          }
1236  
1237          // Set the Content-Length if we have a body or if request is POST/PUT
1238          if ($body || $this->method == self::POST || $this->method == self::PUT) {
1239              $this->setHeaders(self::CONTENT_LENGTH, strlen($body));
1240          }
1241  
1242          if (isset($mbIntEnc)) {
1243              mb_internal_encoding($mbIntEnc);
1244          }
1245  
1246          return $body;
1247      }

1249      /**
1250       * Helper method that gets a possibly multi-level parameters array (get or
1251       * post) and flattens it.
1252       *
1253       * The method returns an array of (key, value) pairs (because keys are not
1254       * necessarily unique. If one of the parameters in as array, it will also
1255       * add a [] suffix to the key.
1256       *
1257       * This method is deprecated since Zend Framework 1.9 in favour of
1258       * self::_flattenParametersArray() and will be dropped in 2.0
1259       *
1260       * @deprecated since 1.9
1261       *
1262       * @param  array $parray    The parameters array
1263       * @param  bool  $urlencode Whether to urlencode the name and value
1264       * @return array
1265       */
1266      protected function _getParametersRecursive($parray, $urlencode = false)
1267      {
1268          // Issue a deprecated notice
1269          trigger_error("The " .  __METHOD__ . " method is deprecated and will be dropped in 2.0.",
1270              E_USER_NOTICE);
1271  
1272          if (! is_array($parray)) {
1273              return $parray;
1274          }
1275          $parameters = array();
1276  
1277          foreach ($parray as $name => $value) {
1278              if ($urlencode) {
1279                  $name = urlencode($name);
1280              }
1281  
1282              // If $value is an array, iterate over it
1283              if (is_array($value)) {
1284                  $name .= ($urlencode ? '%5B%5D' : '[]');
1285                  foreach ($value as $subval) {
1286                      if ($urlencode) {
1287                          $subval = urlencode($subval);
1288                      }
1289                      $parameters[] = array($name, $subval);
1290                  }
1291              } else {
1292                  if ($urlencode) {
1293                      $value = urlencode($value);
1294                  }
1295                  $parameters[] = array($name, $value);
1296              }
1297          }
1298  
1299          return $parameters;
1300      }

1302      /**

1304       *
1305       * This method will try to detect the MIME type of a file. If the fileinfo
1306       * extension is available, it will be used. If not, the mime_magic
1307       * extension which is deprected but is still available in many PHP setups
1308       * will be tried.
1309       *
1310       * If neither extension is available, the default application/octet-stream

1316      protected function _detectFileMimeType($file)

1322              if (self::$_fileInfoDb === null) {
1323                  self::$_fileInfoDb = @finfo_open(FILEINFO_MIME);
1324              }
1325
