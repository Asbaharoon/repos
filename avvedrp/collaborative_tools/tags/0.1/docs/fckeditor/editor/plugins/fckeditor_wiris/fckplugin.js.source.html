 190  function wrs_int_doubleClickHandler(iframe, element) {
 191      if (element.nodeName.toLowerCase() == 'img') {
 192          if (parent.wrs_containsClass(element, 'Wirisformula')) {
 193              if (!parent._wrs_int_window_opened) {
 194                  parent._wrs_int_wirisProperties = {
 195                      'bgColor': FCK.Config['wirisimagebgcolor'],
 196                      'symbolColor': FCK.Config['wirisimagesymbolcolor'],
 197                      'transparency': FCK.Config['wiristransparency'],
 198                      'fontSize': FCK.Config['wirisimagefontsize'],
 199                      'numberColor': FCK.Config['wirisimagenumbercolor'],
 200                      'identColor': FCK.Config['wirisimageidentcolor']
 201                  };
 202                  
 203                  parent._wrs_temporalImage = element;
 204                  wrs_int_openExistingFormulaEditor(iframe);
 205              }
 206              else {
 207                  parent._wrs_int_window.focus();
 208              }
 209          }
 210          else if (parent.wrs_containsClass(element, 'Wiriscas')) {
 211              if (!parent._wrs_int_window_opened) {
 212                  parent._wrs_temporalImage = element;
 213                  wrs_int_openExistingCAS(iframe);
 214              }
 215              else {
 216                  parent._wrs_int_window.focus();
 217              }
 218          }
 219      }
 220  }

 263  function wrs_int_mousedownHandler(iframe, element) {

 265

 268              parent._wrs_int_temporalImageResizing = element;
