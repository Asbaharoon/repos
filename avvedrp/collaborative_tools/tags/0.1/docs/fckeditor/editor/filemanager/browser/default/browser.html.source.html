 118  oConnector.CheckError = function( responseXml )
 119  {
 120      var iErrorNumber = 0 ;
 121      var oErrorNode = responseXml.SelectSingleNode( 'Connector/Error' ) ;
 122  
 123      if ( oErrorNode )
 124      {
 125          iErrorNumber = parseInt( oErrorNode.attributes.getNamedItem('number').value, 10 ) ;
 126  
 127          switch ( iErrorNumber )

 129              case 0 :
 130                  break ;

 132                  alert( oErrorNode.attributes.getNamedItem('text').value ) ;
