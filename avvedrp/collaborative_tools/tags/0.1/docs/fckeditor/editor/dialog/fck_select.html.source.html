 103          if ( sValue.length == 0 ) sValue = sText ;

 105          var oOption = AddComboOption( oActiveEl, sText, sValue, oDOM ) ;
 106  
 107          if ( sValue == GetE('txtSelValue').value )
 108          {
 109              SetAttribute( oOption, 'selected', 'selected' ) ;
 110              oOption.selected = true ;
 111          }
 112      }
