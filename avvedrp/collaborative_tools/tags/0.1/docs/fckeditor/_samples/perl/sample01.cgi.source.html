  39      # Win Apache
  40      if($ENV{'WINDIR'} ne "") { $www_server_os= 'win'; }
  41      if($www_server_os=~ /win/i) { return(1); }
  42      return(0);

 101  sub GetServerPath
 102  {
 103  my $dir;
 104  
 105      if($DefServerPath) {
 106          $dir = $DefServerPath;
 107      } else {
 108          if($ENV{'PATH_INFO'}) {
 109              $dir  = $ENV{'PATH_INFO'};
 110          } elsif($ENV{'FILEPATH_INFO'}) {
 111              $dir  = $ENV{'FILEPATH_INFO'};
 112          } elsif($ENV{'REQUEST_URI'}) {
 113              $dir  = $ENV{'REQUEST_URI'};
 114          }
 115      }
 116      return($dir);
 117  }
